二、模組一：Anchor Score 差異化重構
2.1 問題根因
目前 Anchor Engine 所有錨點的 Score 均顯示 100，代表評分邏輯沒有對不同品質的錨點做出區分。Score 失去意義，使用者無法從中獲取有效的優先級資訊。
2.2 新評分公式
最終分數計算方式
FinalScore = 時間衰減分(25%) + 突破懲罰分(20%) + 時框共振分(25%) + 成交量確認分(15%) + 觸碰頻率分(15%)
分數範圍：0 ~ 100，低於 60 分的錨點降格顯示（灰色）
維度一：時間衰減分（25 分滿分）
邏輯說明：一個錨點上次被觸碰距今越久，其即時參考性越低，應給予扣分。

// 程序員實作參考（JavaScript）
function calcTimeScore(lastTouchTimestamp) {
  const days = (Date.now() - lastTouchTimestamp) / 86400000;
  if (days <= 7)  return 25;
  if (days <= 14) return 18;
  if (days <= 30) return 10;
  return 0;
}

function calcTimeframeScore(timeframes) {
  // timeframes 是一個陣列，例如 ["1h", "4h", "1d"]
  const count = timeframes.length;
  if (count >= 4) return 25;
  if (count === 3) return 18;
  if (count === 2) return 10;
  return 3;
}

function calcVolumeScore(touchVolume, avgVolume20) {

三、模組二：波動預警方向性升級
3.1 問題根因
現有波動預警只顯示「N 根 K 棒」的橫盤狀態，缺少三個關鍵資訊：方向性（上升三角還是下降三角）、成交量狀態（是否萎縮）、距關鍵錨點的距離。
3.2 新增計算邏輯
邏輯一：橫盤型態識別
判定規則
計算過去 N 根 K 棒的最高點序列和最低點序列，分別做線性迴歸
上升三角：高點持平（斜率 < 0.0005），低點上升（斜率 > 0.001）
下降三角：低點持平（斜率 > -0.0005），高點下降（斜率 < -0.001）
對稱三角：高點下降 + 低點上升
矩形整理：高低點均持平（斜率都在 ±0.0005 之間）
function identifyPattern(highs, lows) {
  const highSlope = linearRegression(highs).slope;
  const lowSlope  = linearRegression(lows).slope;
  if (Math.abs(highSlope) < 0.0005 && lowSlope > 0.001)
    return { type: "ascending_triangle", breakDirection: "up" };
  if (Math.abs(lowSlope) < 0.0005 && highSlope < -0.001)
    return { type: "descending_triangle", breakDirection: "down" };
  if (highSlope < -0.001 && lowSlope > 0.001)
    return { type: "symmetrical_triangle", breakDirection: "neutral" };
  return { type: "rectangle", breakDirection: "neutral" };
}
邏輯二：成交量萎縮判定
function isVolumeContracted(volumes, windowSize = 10) {
  const recent = volumes.slice(-windowSize);
  const baseline = volumes.slice(-windowSize * 3, -windowSize);
  const recentAvg   = recent.reduce((a,b)=>a+b,0) / recent.length;
  const baselineAvg = baseline.reduce((a,b)=>a+b,0) / baseline.length;
  return {
    contracted: recentAvg < baselineAvg * 0.7,  // 萎縮超過 30%
    ratio: (recentAvg / baselineAvg * 100).toFixed(1) + "%"
  };
}
邏輯三：距錨點距離計算
function getAnchorDistance(currentPrice, anchors) {
  const nearestAbove = anchors
    .filter(a => a.price > currentPrice)
    .sort((a,b) => a.price - b.price)[0];
  const nearestBelow = anchors
    .filter(a => a.price < currentPrice)
    .sort((a,b) => b.price - a.price)[0];
  return {
    resistance: nearestAbove ? {
      price: nearestAbove.price,
      distPct: ((nearestAbove.price - currentPrice)/currentPrice*100).toFixed(2)
    } : null,
    support: nearestBelow ? {
      price: nearestBelow.price,
      distPct: ((currentPrice - nearestBelow.price)/currentPrice*100).toFixed(2)
    } : null
  };
}

function passThreshold(symbol, sizeUSD) {
  const threshold = THRESHOLDS[symbol] || THRESHOLDS.DEFAULT;

}
第二關：持續性驗證（3 分鐘計時）
// 進入觀察池：第一次看到且金額達標
// 持續計時：每次收到該價位更新就刷新 lastSeen
// 升格顯示：firstSeen 距今超過 3 分鐘 → 移入已確認列表
// 清除：超過 30 秒沒有收到該價位更新 → 視為撤單，移除
const PERSIST_MS = 3 * 60 * 1000;  // 3 分鐘
const STALE_MS   = 30 * 1000;       // 30 秒無更新視為消失
function checkPersistence(order, now) {
  if (now - order.lastSeen > STALE_MS) return "REMOVED";
  if (now - order.firstSeen >= PERSIST_MS) return "CONFIRMED";
  return "PENDING";
}
第三關：OKX 合約單位換算（重要！）
OKX 的掛單量單位是「張數」而非直接的幣數，不換算會導致金額計算錯誤。
// OKX 各合約每張代表的幣數
const OKX_CONTRACT_SIZE = {
  "BTC-USDT-SWAP": 0.01,   // 每張 = 0.01 BTC
  "ETH-USDT-SWAP": 0.1,    // 每張 = 0.1 ETH
  "SOL-USDT-SWAP": 1,      // 每張 = 1 SOL
};
function convertOKXQty(instId, contracts, markPrice) {
  const size = OKX_CONTRACT_SIZE[instId] || 1;
  const coinQty = contracts * size;
  return coinQty * markPrice;  // 轉換為美元
}
4.3 多交易所交叉驗證
信心等級判定規則
HIGH（金色）：同一價位（容差 ±$50）在 3 家以上交易所同時出現確認掛單
MED（藍色）：同一價位在 2 家交易所同時出現確認掛單
LOW（灰色）：只在單一交易所出現的確認掛單
const CROSS_PRICE_TOLERANCE = 50;  // $50 價格容差
function calcCrossExchangeScore(targetOrder, allConfirmed) {
  const matches = allConfirmed.filter(o =>
    o.id !== targetOrder.id &&
    o.side === targetOrder.side &&
    o.exchange !== targetOrder.exchange &&
    Math.abs(o.price - targetOrder.price) <= CROSS_PRICE_TOLERANCE
  );
  const uniqueExchanges = [...new Set(matches.map(o => o.exchange))];
  if (uniqueExchanges.length >= 2) return "HIGH";
  if (uniqueExchanges.length === 1) return "MED";
  return "LOW";
}

5.2 計算方式
function calcDistanceFromPrice(anchorPrice, currentPrice) {
  const pct = (anchorPrice - currentPrice) / currentPrice * 100;
  const sign = pct >= 0 ? "+" : "";
  const direction = pct >= 0 ? "↑ 上方" : "↓ 下方";
  return {
    pct: sign + pct.toFixed(2) + "%",
    direction,
    isAbove: pct >= 0
  };
}

六、模組五：板塊輪動與個幣候選橋接
6.1 功能說明
在「熱門板塊漲幅比較」頁面中，點入某個強勢板塊後，系統自動顯示「該板塊中符合以下條件的交易候選幣種」：
進入候選清單的條件（三個同時滿足）
條件 A：該幣種有 Score ≥ 70 的支撐錨點，且距現價 1% ~ 8% 之內
條件 B：波動預警橫盤 K 棒數 ≥ 20 根（代表已蓄力一段時間）
條件 C：成交量處於萎縮狀態（近期成交量 < 20根均量的 70%）
6.2 候選排序邏輯
function scoreCandidates(candidates) {
  return candidates.map(c => {
    let score = 0;
    // Anchor 品質（最重要）
    score += c.anchorScore * 0.40;
    // 橫盤蓄力程度（K棒數越多越好，上限 80根）
    score += Math.min(c.consolidationBars / 80 * 100, 100) * 0.30;
    // 距支撐距離（越近越好，最佳 0.5%~3%）
    const distScore = c.distPct <= 3 ? 100 : Math.max(0, 100 - (c.distPct - 3) * 20);
    score += distScore * 0.20;
    // 成交量萎縮程度（萎縮越多越好）
    score += (1 - c.volumeRatio) * 100 * 0.10;
    return { ...c, candidateScore: Math.round(score) };
  }).sort((a, b) => b.candidateScore - a.candidateScore);
}
